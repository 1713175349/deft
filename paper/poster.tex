  \headerbox{Classical DFT}{name=cdft,column=0,row=0}{ Classical
    Density-Functional Theory is a continuum empirical approach to
    modelling fluid behavior when the density is inhomogeneous---as is
    the case at interfaces.  We have developed a classical density
    functional for water, which provides a description of hydrophobic
    interactions.  }

  \headerbox{Statistical Associating Fluid
    Teory}{name=saft,column=0,below=cdft}{ The statistical associating
    fluid theory (SAFT) is a well-established approach for the
    description of the thermodynamics and phase equilibria of fluid
    systems, with particular applicability to hydrogen-bonding fluids.
    Our functional reduces in the homogeneous limit to the SAFT model
    for water developed in reference~\cite{clark2006developing}.
    \begin{center}
      \includegraphics[width=\columnwidth]{figs/pressure-with-isotherms}
    \end{center}
    This functional accurately describes the liquid-vapor phase
    diagram away from the critical point, but gives a poor description
    of the compressibility.
  }

  \headerbox{The
    Functional}{name=fitting,column=0,below=saft,above=bottom}{ We
    introduce a new classical density functional for water.  The
    Helmholtz free energy functional is based on SAFT, and is composed
    of the usual terms:
    \begin{equation}
      F[n] = F_\textit{id}[n] + F_\textit{hs}[n] + F_\textit{assoc}[n] + F_\textit{disp},
    \end{equation}
    We model the repulsive interactions using the White Bear version
    of the Fundamental-Measure Theory~(FMT)
    functional~\cite{roth2002whitebear}.  The association term is
    taken from~\cite{yu2002fmt-dft-inhomogeneous-associating}, and we
    used a simple and efficient functional for the dispersion
    interaction with a single tunable parameter.
  }
